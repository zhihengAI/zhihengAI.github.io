new_state_dict = {}

# 遍历原来的 state_dict 并去掉 'yyy.xxx' 前缀
for k, v in state_dict.items():
    new_key = k.replace('yyyyyyy.xxxxx.', '')
    new_state_dict[new_key] = v

# 同样是保存，这个 final.pt 就是这个开源模型训练好的最终权重参数了
torch,save(selected_params, 'QWen-Audio-Encoder-final.pt')
